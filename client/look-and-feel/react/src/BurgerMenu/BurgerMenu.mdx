# Burger-Menu

## How to use

To use the BurgerMenu component import it like that:

```tsx
import { BurgerMenu } from "@axa-fr/react-toolkit-burger-menu";
import error from "@material-symbols/svg-400/outlined/error.svg";
import { Svg } from "./Svg";

<BurgerMenu
  items={[
    {
      label: "test",
      isDisabled: false,
      icon: <Svg src={error} />,
    },
    {
      label: "test",
      isDisabled: false,
      icon: <Svg src={error} />,
    },
  ]}
/>;
```
