# Navigation Bar Item 

## Default Navigation Bar Item

<Canvas of={NavBarItemStories.Default} />

### Import

```tsx
import { NavBarItem  } from "@axa-fr/design-system-react/agent";
```

### Use

```tsx
const NavBarItemDefault = () => (
  <ul style={{ listStyle: 'none' }}>
    <NavBarItem
      style={{ width: '100px' }}
      actionElt={
        <a
          className="af-nav__link"
          href="https://axa.fr"
          onClick={(e) => console.log(e)}>
          HomeLINK
        </a>
      }
    />
  </ul>
);
export default NavBarItemDefault;
```
