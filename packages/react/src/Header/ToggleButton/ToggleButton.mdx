# Toggle Button

## Stories

<Canvas of={ToggleButtonStories.Default} />

## Import

```tsx
import { ToggleButton } from "@axa-fr/design-system-react/agent";
```

## Use

```tsx
const actionToggle = {
  icon: "menu-hamburger",
  title: "Toggle",
};

const ToggleButtonLayout = () => (
  <ToggleButton idControl="mainmenu">
    <Action
      id="togglemenu"
      icon={actionToggle.icon}
      title={actionToggle.title}
    />
  </ToggleButton>
);
export default ToggleButtonLayout;
```
