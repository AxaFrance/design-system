# Tabs

## How to use

To use the tabs component import it like that:

```tsx title="test.js"
import { Tabs } from "@axa-fr/design-system-react/client";
import accessTime from "@material-symbols/svg-400/outlined/access_time.svg";
import { Svg } from "../../Svg";

const MyComponent = () => (
  <Tabs
    items={[
      { title: "Tab 1", content: "Content 1" },
      { title: "Tab 2", content: "Content 2", icon: <Svg src={accessTime} /> },
      { title: "Tab 3", content: "Content 3" },
    ]}
    direction={Direction.center}
    preSelectedTabIndex={0}
  />
);
```
