# ContentTabList

## Import

```tsx
import { ContentTabList } from "@axa-fr/design-system-react/client";
```

## Use

```tsx

  <ContentTabList
    items={[
      {
        id: "1",

      },
      {
        id: "2",
        title: "Remboursement soins",
        tag: "En attente",
        onDownload,
        onDisplay,
      },
    ]}
  />
);
```
