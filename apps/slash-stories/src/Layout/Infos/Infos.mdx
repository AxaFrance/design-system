# Infos

> **Deprecated prop**: The `classModifier` prop is deprecated. Use `className` to add custom CSS modifier classes instead.

## Import

```tsx
import { Infos } from "@axa-fr/canopee-react/distributeur";
```

## Use

```tsx
const infos = [
  { word: "Customer :", definition: "0123456789 - NOM" },
  { word: "Wallet:", definition: "000123456789" },
  { word: "Status :", definition: "New business" },
];

const InfosComponent = () => <Infos infos={infos} />;
```
