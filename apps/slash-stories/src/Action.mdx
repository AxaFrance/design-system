## Action

> **Deprecated prop**: The `classModifier` prop is deprecated. Use `className` to add custom CSS modifier classes instead.

### Import

```tsx
import { Action } from "@axa-fr/canopee-react/distributeur";
```

### Use

#### Link Action Button Example:

```tsx
export const LinkAction = () => (
  <Action
    id="id"
    icon="link"
    href="http://www.axa.fr"
    title="Test Link"
    target="_blank"
  />
);
```
