# ItemLabel

To use the ItemLabel import it like that:

```tsx
import { ItemLabel } from "@axa-fr/design-system-look-and-feel-react";
//or
import { ItemLabel } from "@axa-fr/design-system-apollo-react/lf";

const MyComponent = () => (
  <ItemLabel
    label="Label Text"
    description="Description Text"
    required
    inputId="input-id"
    idLabel="label-id"
    sideButtonLabel="Side Button"
    onSideButtonClick={handleClickSideButton}
    buttonLabel="ButtonLabel"
    onButtonClick={handleClickButton}
  />
);
```
