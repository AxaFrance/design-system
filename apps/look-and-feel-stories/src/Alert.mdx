## Alert

### Import

```tsx title="test.js"
import { Alert } from "@axa-fr/design-system-look-and-feel-react/";
```

### Use

```tsx title="test.js"
export const AlertComponent = () => (
  <Alert type="information" title="My alert title">
    My alert content
  </Alert>
);
```
