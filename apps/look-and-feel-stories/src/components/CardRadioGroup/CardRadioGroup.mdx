# CardRadioGroup components

## How to use

To use the `CardRadioGroup`component import it like this:

```tsx
import { CardRadioGroup } from "@axa-fr/canopee-react/client";

const MyComponent = () => (
  <CardRadioGroup
    cardStyle="vertical"
    name="city"
    label="Choisissez une ville"
    options={[
      {
        label: "Paris",
        value: "paris",
        icon: homeIcon,
        description: "Capitale de la France",
        subtitle: "Nord",
      },
      {
        label: "Bruxelles",
        value: "bruxelles",
        icon: homeIcon,
        description: "Capitale de la Belgique",
      },
      {
        label: "Lille",
        value: "lille",
        icon: homeIcon,
        checked: true,
      },
    ]}
  />
);
```
