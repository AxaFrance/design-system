# Header Client

## Import

```tsx
import { Header } from "@axa-fr/design-system-look-and-feel-react";
```

## Use

### Use Default Header

```tsx
const DefaultHeader = () => <Header />;
```

### Use Header with Navigation Bar

```tsx
const HeaderWithNavBar = () => <Header links={links} />;
```

### Use Header with Navigation Bar and right item

```tsx
const HeaderWithNavBarAndRightItemTitle = () => (
  <Header links={links} rightItem="AXA Banque" />
);
```

### Use Header with Navigation Bar, right item and previous link

```tsx
const HeaderWithLinksAndRightItemAndButton = () => (
```
