# CardRadio components

## How to use

To use the `CardRadio`component import it like this:

```tsx
import { CardRadio } from "@axa-fr/canopee-react/prospect";

const MyComponent = () => (
  <CardRadio
    type="vertical"
    name="city"
    label="Choisissez une ville"
    options={[
      {
        label: "Paris",
        value: "paris",
        icon: homeIcon,
        description: "Capitale de la France",
        subtitle: "Nord",
      },
      {
        label: "Bruxelles",
        value: "bruxelles",
        icon: homeIcon,
        description: "Capitale de la Belgique",
      },
      {
        label: "Lille",
        value: "lille",
        icon: homeIcon,
        checked: true,
      },
    ]}
  />
);
```
