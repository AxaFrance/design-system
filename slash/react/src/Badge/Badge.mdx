## Badge

### Import

```tsx title="test.js"
import { Badge } from "@axa-fr/design-system-slash-react/";
```

### Use

```tsx title="test.js"
export const SimpleBadge = () => (
  <Badge classModifier="success">Lorem ipsum</Badge>
);
```

### With Icon

```tsx title="test.js"
export const WithIconBadge = () => (
  <Badge classModifier="error">
    <i className="glyphicon glyphicon-bell" />
  </Badge>
);
```
