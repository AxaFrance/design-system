# Card

### Import

```tsx
import { Card } from "@axa-fr/design-system-react/client";
```

### Use

```tsx
export const CardComponent = () => (
  <Card>
    <h1 className="af-card__title">My card title</h1>
    <p>My card content</p>
  </Card>
);
```
