# ClickList

## Import

```tsx
import { ClickList } from "@axa-fr/design-system-slash-react/client";
```

## Use

### Use with left icon

```tsx
const MyComponentWithIcon = () => (
  <ClickList
    items={[
      {
        id: "report-claim",
        icon: <Svg src={error} />,
        label: "Déclarer un sinistre",
      },
      {
        id: "follow-claim",
        icon: <Svg src={contentPasteGo} />,
        label: "Suivre mes sinistres",
        disabled: true,
      },
    ]}
  />
);
```
