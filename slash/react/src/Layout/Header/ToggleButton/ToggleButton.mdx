# Toggle Button

## Import

```tsx
import { ToggleButton } from "@axa-fr/design-system-slash-react/slash";
```

## Use

```tsx
const actionToggle = {
  icon: "menu-hamburger",
  title: "Toggle",
};

const ToggleButtonLayout = ({
  children,
  idControl,
}: TToggleButtonLayoutProps) => (
  <ToggleButton idControl={idControl}>{children}</ToggleButton>
);
export default ToggleButtonLayout;
```
